# 蓝桥杯刷题挑战赛

[题目链接](https://www.lanqiao.cn/contests/lqcup1/challenges/)

A.明码

```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
    int n, m;
    char map[20][20];
    for (int m = 1; m <= 10; m++)
    {
        memset(map, (int)' ', sizeof(map));
        for (int i = 1; i <= 16; i++)
        {
            cin >> m >> n;
            int cnt = 8; // 从map的8，16号位开始填墨迹
            while (cnt >= 0)
            {
                if (m & 1)
                    map[i][cnt] = '*';
                if (n & 1)
                    map[i][8 + cnt] = '*';
                cnt--;
                m >>= 1;
                n >>= 1;
            }
        }
        for (int i = 0; i < 20; i++)
        {
            for (int j = 0; j < 20; j++)
                cout << map[i][j];
            cout << '\n';
        }
    }
    // cout << (ll)pow(9, 9);//结果为9^9
    return 0;
}
```
